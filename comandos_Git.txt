git init                            //Iniciar git
git add _nombre de archivo_         // enviar al repositorio cambios de un archivo
git add .                           // enviar cambios al repositorio de todos los archivos de la carpeta
git commit -m "..."                 // guardar los cambios
git code                            // abri code
git code _nombre de archivo_        // abrir documento en code
git log _nombre de archivo_         // 
git log --stat                      //Ver los bites que se usaron para el cambio
git show
git ls
git ls -al
git status 

clear
_comando: crtl l

git reset _codigo de version_ --hard //regresar a una version anterior y eliminar todo lo posterior 
                                        a esa version
git reset _codigo de version_ --sofd //regresar a una version a anterior
git reset HEAD                       // Este es el comando para sacar archivos del área de Staging. 
                                        No para borrarlos ni nada de eso, solo para que los últimos 
                                        cambios de estos archivos no se envíen al último commit, 
                                        a menos que cambiemos de opinión y los incluyamos 
                                        de nuevo en staging con git add, por supuesto.

git checkout _codigo_ _archivo_      //regresar a una version anterior
git checkout master _archivo_

git rm                              //elimina archivos de Git sin eliminar su historial del sistema de versiones
git rm --cached                     //Elimina los archivos del área de Staging y del próximo commit pero los mantiene en nuestro disco duro.
git rm --force                      //Elimina los archivos de Git y del disco duro.
